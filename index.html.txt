</head>
<body style="margin:0; overflow:hidden;">

<!-- スマホライブラリから選択 -->
<input type="file" id="imageInput" accept="image/png, image/jpeg" style="position:absolute; top:10px; left:10px; z-index:10;">

<script>
let camera, scene, renderer;
let imagePlane;

init();
animate();

function init() {
    scene = new THREE.Scene();

    camera = new THREE.PerspectiveCamera(
        70,
        window.innerWidth / window.innerHeight,
        0.01,
        20
    );

    renderer = new THREE.WebGLRenderer({ alpha: true });
    renderer.setSize(window.innerWidth, window.innerHeight);
    renderer.xr.enabled = true;
    document.body.appendChild(renderer.domElement);

    // 初期プレースホルダー画像
    const placeholderTexture = new THREE.TextureLoader().load(""); // 空でもOK
    const geometry = new THREE.PlaneGeometry(1, 1); // 比率は後で変更可能
    const material = new THREE.MeshBasicMaterial({
        map: placeholderTexture,
        transparent: true
    });

    imagePlane = new THREE.Mesh(geometry, material);
    imagePlane.position.set(0, 0, -2);
    scene.add(imagePlane);

    // スマホライブラリ選択時の処理
    const input = document.getElementById("imageInput");
    input.addEventListener("change", function(event) {
        const file = event.target.files[0];
        if (!file) return;
        const reader = new FileReader();
        reader.onload = function(e) {
            const texture = new THREE.TextureLoader().load(e.target.result, () => {
                // 読み込み後に比率を反映
                const img = new Image();
                img.src = e.target.result;
                img.onload = function() {
                    const ratio = img.width / img.height;
                    imagePlane.geometry.dispose();
                    imagePlane.geometry = new THREE.PlaneGeometry(ratio, 1);
                }
            });
            imagePlane.material.map = texture;
            imagePlane.material.needsUpdate = true;
        };
        reader.readAsDataURL(file);
    });

    window.addEventListener("resize", onWindowResize);
}

function onWindowResize() {
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
}

function animate() {
    renderer.setAnimationLoop(render);
}

function render() {
    if (imagePlane) {
        // 常にカメラ正面
        imagePlane.quaternion.copy(camera.quaternion);
    }
    renderer.render(scene, camera);
}
</script>
